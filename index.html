import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

# Dados de exemplo
X = np.random.rand(100, 1) * 10
y = 2 * X.flatten() + 1 + np.random.randn(100) * 0.5

# Dividir dados
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

# Treinar modelo
model = LinearRegression()
model.fit(X_train, y_train)

# Fazer predições
predictions = model.predict(X_test)
print(f"Score: {model.score(X_test, y_test):.2f}")

⚡ FastAPI

Framework moderno para APIs com performance alta e documentação automática. Ideal para microserviços.

fastapi_example.py
from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI()

class Item(BaseModel):
    name: str
    price: float
    is_offer: bool = False

@app.get("/")
def read_root():
    return {"Hello": "World"}

@app.get("/items/{item_id}")
def read_item(item_id: int, q: str = None):
    return {"item_id": item_id, "q": q}

@app.post("/items/")
def create_item(item: Item):
    return {"message": f"Item {item.name} created!"}

🍃 Flask

Micro-framework minimalista e flexível. Excelente para projetos pequenos e médios ou quando você quer controle total.

flask_example.py
from flask import Flask, render_template, request, jsonify

app = Flask(__name__)

@app.route('/')
def home():
    return render_template('index.html')

@app.route('/api/users', methods=['GET', 'POST'])
def users():
    if request.method == 'POST':
        user_data = request.get_json()
        return jsonify({"message": "User created", "data": user_data})
    else:
        return jsonify({"users": ["Alice", "Bob", "Charlie"]})

if __name__ == '__main__':
    app.run(debug=True)

🎮 Pygame

Biblioteca para desenvolvimento de jogos 2D. Perfeita para aprender programação através de jogos divertidos.

pygame_example.py
import pygame
import sys

# Inicializar Pygame
pygame.init()

# Configurações
WIDTH, HEIGHT = 800, 600
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Meu Primeiro Jogo")

# Cores
WHITE = (255, 255, 255)
RED = (255, 0, 0)

# Posição do jogador
player_x, player_y = WIDTH // 2, HEIGHT // 2

# Loop principal
clock = pygame.time.Clock()
running = True

while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
    
    # Controles
    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT]:
        player_x -= 5
    if keys[pygame.K_RIGHT]:
        player_x += 5
    
    # Desenhar
    screen.fill(WHITE)
    pygame.draw.circle(screen, RED, (player_x, player_y), 20)
    pygame.display.flip()
    clock.tick(60)

pygame.quit()
sys.exit()
